waste_handler = {	
	category = worker
	condition_string = WORKER_JOB_TRIGGER
	building_icon = eco_building_waste_recovery
	clothes_texture_index = 2

	possible = {
		worker_job_check_trigger = yes
		owner = { is_fallen_empire = no }
	}

	resources = {
		category = planet_waste
		produces = {
			trigger = {
				owner = { has_policy_flag = eco_policy_material }
			}
			minerals = 1
		}
		produces = {
			trigger = {
				owner = { has_policy_flag = eco_policy_default }
			}
			energy = 1
		}
		produces = {
			trigger = { 
				owner = { has_valid_civic = civic_environmentalist }
			}
			unity = 1
		}
	}

	planet_modifier = {
		planet_amenities_add = 1
	}
	
	triggered_planet_modifier = {
		potential = { owner = { has_policy_flag = eco_policy_reuse } }
		modifier = {
			planet_amenities_add = 1
		}
	}
	
	triggered_planet_modifier = {
		potential = { owner = { has_valid_civic = eco_civic_scrapper } }
		modifier = {
			trade_value_add = 1
		}
	}
	
	triggered_planet_modifier = {
		potential = {
			OR = {
				has_trait = trait_robot_domestic_protocols
				has_trait = trait_charismatic
			}
		}
		modifier = {
			planet_amenities_add = 0.2
		}	
	}
	
	triggered_planet_modifier = {
		potential = {
			has_trait = trait_repugnant
		}
		modifier = {
			planet_amenities_add = -0.2
		}
	}

	triggered_planet_modifier = {
		potential = {
			OR = {
				has_trait = trait_robot_domestic_protocols
				has_trait = trait_charismatic
			}
			owner = { has_policy_flag = eco_policy_reuse }
		}
		modifier = {
			planet_amenities_add = 0.2
		}	
	}
	
	triggered_planet_modifier = {
		potential = {
			has_trait = trait_repugnant
			owner = { has_policy_flag = eco_policy_reuse }
		}
		modifier = {
			planet_amenities_add = -0.2
		}
	}
	
	weight = {
		weight = @worker_job_weight
		modifier = {
			factor = 10
			is_enslaved = yes
			can_take_servant_job = no
		}	
		modifier = {
			factor = 2
			OR = {	
				is_non_sapient_robot = yes
				is_shackled_robot = yes
			}
			can_take_servant_job = no
			owner = { has_technology = tech_droid_workers }
		}	
		modifier = {
			factor = 10
			OR = {	
				is_non_sapient_robot = yes
				is_shackled_robot = yes
			}
			can_take_servant_job = no
			owner = { NOT = { has_technology = tech_droid_workers } }
		}		
		modifier = {
			factor = 2
			species = {
				has_species_flag = racket_species_flag
			}
		}		
		modifier = {
			factor = 1.2
			has_trait = trait_very_strong
		}
		modifier = {
			factor = 1.1
			has_trait = trait_strong
		}
		modifier = {
			factor = 1.1
			has_trait = trait_nerve_stapled
		}		
		modifier = {
			factor = 0.9
			has_trait = trait_weak
		}					
	}
}
